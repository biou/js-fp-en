function goOutTonight() {
  return new Promise(function (resolve, reject) {
    $.ajax('http://meteolux.lu/tonight')
      .done(function (weather) {
        if (! weather.rain) resolve()
        else reject(new Error("it's raining"))
      })
      .fail(reject)
  }
}
