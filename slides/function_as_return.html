function transpose (interval) {
  return function (note) {
    return note + interval;
  }
}

var octave = transpose(8);
octave(1) //  9
octave(2) // 10



function transpose (type) {
  switch (type) {
    case 'octave': return function (note) {
                            return note + 8;
                          }
    case 'note': return function (note) {
                          return note + 1;
                        }
  }
}
