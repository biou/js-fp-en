//non functional example

var notes = [ 1, 2, 3 ]

function displayNotes(notes, elem) {
  var buffer = ''
  for (var i = 0; i < notes.length; i++) {
    buffer += noteValueToText(notes[i]) + ' '
  }
  document.getElementById(elem).innerHTML = buffer
  // displays "D E F"
}

function noteValueToText(note) {
  var notes = [ 'C', 'D', 'E', 'F' , 'G', 'H', 'I']
  return notes[note % 7]
}
