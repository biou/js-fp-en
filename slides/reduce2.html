reduce()

Réduit une liste, en appliquant une fonction à chaque
élément, et retourne le résultat final. 


var liste = [ -1, -2, 3, 4 ]

function positif(acc, e) {
  return acc + (e >= 0 ? 1 : 0)
}

liste.reduce(positif, 0) // 2
Utile pour transformer une liste en une valeur finale: somme, moyenne, etc.
Ou compter les éléments qui vérifient une condition
